188  void PipelineController::computeAll()
189  {
190 
191  for (size_t iData = 0; iData < myConfig["readers"].size(); iData++)
192  {
193  // set diff result path for each input file
194  auto inputName = fs::path(myConfig["readers"][iData]["dataFileName"].get<std::string>());
195  std::string storagePath = Context::dataStorage().generateStoragePath(inputName.stem().string());
196  Context::dataStorage().setStoragePath(storagePath);
197 #if PPP_BUILD_TYPE
198  std::shared_ptr<Processor> reader = createProcessor(myConfig["readers"][iData]);
199 #else
200  std::shared_ptr<Reader> reader = std::make_shared<Reader>();
201 #endif
202  reader->setConfig(myConfig["readers"][iData]);
203  reader->process();
204  reader->prepareOutput();
205  auto data = reader->outputData();
206  if (!data)
207  {
208  std::cout << "Failed to load the data from file: \n"
209  << " Here is the reader config: \n"
210  << myConfig["readers"][iData][""] << std::endl;
211  }
212 
213  auto info = std::make_shared<Information>();
214  (*info)["parallelism"] = myConfig["parallelism"];
215 
216  compute(data, info);
217 
218  // report writing, todo: get output name from configuration
219  std::string outfile = Context::dataStorage().getFullPath("processed_info.json");
220  std::fstream report_info(outfile);
221 
222  // processed data writing, writer is not always needed
223  bool hasWriter = !myConfig["writers"].empty();
224  if (hasWriter)
225  {
226 #if PPP_BUILD_TYPE
227  std::shared_ptr<Processor> writer = createProcessor(myConfig["writers"][iData]);
228 #else
229  std::shared_ptr<Writer> writer = std::make_shared<Writer>();
230 #endif
231  writer->setConfig(myConfig["writers"][iData]);
232  if (myProcessors.size() == 0) // if there is no processor, do data IO format translation
233  {
234  writer->setInputData(data);
235  writer->setInputInformation(info);
236  }
237  else
238  {
239  writer->setInputData(myProcessors[myProcessors.size() - 1]->outputData());
240  writer->setInputInformation(myProcessors[myProcessors.size() - 1]->getOutputInformation());
241  }
242  writer->prepareInput();
243  writer->process();
244  }
245  report_info << (*info);
246  }

279  void PipelineController::compute(std::shared_ptr<DataObject> data, std::shared_ptr<Information> info)
280  {
281 
285  callback_wrapper = std::bind(&PipelineController::handleSignal, this, std::placeholders::_1);
286 #ifndef _WIN32
287  struct sigaction sigIntHandler;
288  sigIntHandler.sa_handler = ppp_signal_callback_handler;
289  sigemptyset(&sigIntHandler.sa_mask); // posix API
290  sigIntHandler.sa_flags = 0;
291  sigaction(SIGINT, &sigIntHandler, NULL); // more signal handler can be added for diff signal
292 #endif
293 
294  size_t nCores = myConfig["parallelism"]["threadsOnNode"];
295  bool serialMode = nCores == 1UL;
296 
298  std::shared_ptr<ThreadPoolType> threadPool;
299  if (not serialMode)
300  {
301  threadPool = std::shared_ptr<ThreadPoolType>(new ThreadPoolType());
302  if (data->itemCount() < nCores)
303  {
304  VLOG_F(LOGLEVEL_DEBUG, "item count is smaller than thread count!, is this a test data?");
305  }
306  }
307 
308 
309  for (std::size_t i = 0; i < myProcessors.size(); i++)
310  {
311  std::string pname = myConfig["processors"][i]["className"];
312  VLOG_F(LOGLEVEL_PROGRESS, " ========processor #%lu %s started=======", i, pname.c_str());
313  myCurrentProcessor = myProcessors[i]; // current processor index is needed in handleSignal()
314  auto start = std::chrono::steady_clock::now();
315  json thisConfig = myConfig["processors"][i];
316  myProcessors[i]->setConfig(thisConfig);
317  if (i == 0)
318  {
319  myProcessors[i]->setInputData(data);
320  myProcessors[i]->setInputInformation(info);
321  }
322  else
323  {
324  myProcessors[i]->setInputData(myProcessors[i - 1]->outputData());
325  myProcessors[i]->setInputInformation(myProcessors[i - 1]->getOutputInformation());
326  }
327  myProcessors[i]->setOperator(myOperator);
328 
329  std::shared_ptr<Executor> aExecutor;
330  if (serialMode)
331  aExecutor = std::make_shared<SequentialExecutor>(myProcessors[i]);
332  else
333  aExecutor = std::make_shared<ThreadPoolExecutor>(myProcessors[i], nCores, threadPool);
334 
335  aExecutor->process(); // must be declared as pointer, otherwise no polymorphism!
336  std::chrono::duration<double, std::milli> duration = std::chrono::steady_clock::now() - start;
337  VLOG_F(LOGLEVEL_PROGRESS, " ====== processor #%lu %s completed in %lf seconds =====", i, pname.c_str(),
338  duration.count() / 1000);
339  }
340  }
