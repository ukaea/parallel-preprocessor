272  void readBrep(std::string filename, const json default_metadata = json())
273  {
274  typedef uint64_t UniqueIdType;
275  json metadata;
276  std::unordered_map<uint64_t, json> gpMap;
277  if (this->myConfig.contains("metadataFileName")
278  && !this->myConfig["metadataFileName"].is_null())
279  {
280  std::string mfile = this->myConfig["metadataFileName"];
281  if (not fs::exists(mfile))
282  {
283  LOG_F(WARNING, "metadataFileName `%s` provided config does not exist!", mfile.c_str());
284  }
285  else
286  {
287  std::ifstream i(mfile);
288  i >> metadata;
289  for (const auto& p : metadata)
290  {
291  gpMap.insert_or_assign(p["uniqueId"].get<UniqueIdType>(), p);
292  // idSet.insert(p["uniqueId"].get<uint64_t>());
293  }
294  }
295  }
296  else
297  {
298  if (default_metadata.is_null())
299  {
300  LOG_F(WARNING, "only shape are read, no meta data (json file) found or provided as default");
301  LOG_F(WARNING, " default geometry metadata like material color, etc should be provided here");
302  }
303  }
304  // read a Compound or CompSolid, return a vector of TopoDS_Shape of Solid type
305 
306  BRep_Builder cBuilder;
307  TopoDS_Shape shape;
308  BRepTools::Read(shape, filename.c_str(), cBuilder);
309  if (shape.IsNull())
310  throw std::runtime_error("brep read data is Null for:" + filename);
311 
312  // explore solids and calc Id
313  int solidCount = 0;
314  TopExp_Explorer Ex(shape, TopAbs_SOLID);
315  while (Ex.More())
316  {
317  const TopoDS_Shape& s = Ex.Current();
318  UniqueIdType gid = OccUtils::uniqueId(s);
319  if (gpMap.size() > 0)
320  {
321  const auto& p = gpMap[gid];
322  bool suppressed = false;
323  if (p.contains("suppressed"))
324  {
325  suppressed = p["suppressed"].get<bool>();
326  }
327  if (not suppressed)
328  registerShape(s, p);
329  else
330  {
331  const char* ns = p["name"].get<std::string>().c_str();
332  LOG_F(WARNING, "shape `%s` has been marked as suppressed, so it will not be loaded ", ns);
333  }
334  }
335  else
336  {
337  json mdata = json{{"name", "geometry"}, {"material", "unknown"}};
338  // if default meta data can be generated, it is possible to read
339  registerShape(s, mdata);
340  }
341  solidCount++; // sequenceID, started with zero
342  Ex.Next();
343  }
