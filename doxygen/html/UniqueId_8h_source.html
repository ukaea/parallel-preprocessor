104  std::set<UniqueIdType> nearbyIds(UniqueIdType id)
105  {
106  std::set<UniqueIdType> ids;
107  std::vector<std::array<UniqueIdType, 3>> items;
108  for (int i = 0; i < ID_ITEM_COUNT; i++)
109  {
110  UniqueIdType s = ROUND_PRECISION_MASK * (1UL << ID_ITEM_BITS * i);
111  items.push_back({id - s, id, id + s}); // overflow and underflow is fine
112  }
113  // generate combination, 3**4, generate ijkl index
114  for (int i = 0; i < 3; i++)
115  {
116  for (int j = 0; j < 3; j++)
117  {
118  for (int k = 0; k < 3; k++)
119  {
120  for (int l = 0; l < 3; l++)
121  {
122  UniqueIdType shift = items[0][i] + items[0][j] + items[0][k] + items[0][l];
123  ids.insert(id + shift);
124  }
125  }
126  }
127  }

156  UniqueIdType uniqueId(const std::vector<double> values)
