13  CollisionType CollisionDetector::calcCollisionType(std::shared_ptr<BRepAlgoAPI_BuilderAlgo> mkGFA,
14  std::vector<Standard_Real> origVols,
15  std::vector<ItemIndexType> ij)
16  {
17  using namespace OccUtils;
18  size_t originalShapeCount = mkGFA->Arguments().Size();
19  const auto& origShapes = mkGFA->Arguments();
20  std::vector<TopoDS_Shape> resultShapes; // output is limited to Solid shape type
21  for (TopExp_Explorer anExp(mkGFA->Shape(), TopAbs_SOLID); anExp.More(); anExp.Next())
22  {
23  resultShapes.push_back(anExp.Current()); // what kind of copy, const ref of the shape is given
24  }
25  if (originalShapeCount != 2)
26  {
27  throw std::runtime_error("this collision type detection method only works for 2 solids");
28  }
29 
30  // resultShapes.size() > 0, otherwise there must be some error during general fusion
31  if (resultShapes.size() == 1)
32  {
33  return CollisionType::Coincidence;
34  }
35  else if (resultShapes.size() >= originalShapeCount)
36  {
37  // size_t modifiedFaceCount = 0;
38  // size_t modifiedEdgeCount = 0;
39  if (mkGFA->HasModified()) // has modified solids after general fuse
40  {
41  if (origVols.size() == 0) // empty volume vector, default parameter value
42  {
43  origVols.clear();
44  for (const auto& s : origShapes)
45  {
46  GProp_GProps v_props;
47  BRepGProp::VolumeProperties(s, v_props);
48  auto v = v_props.Mass();
49  origVols.push_back(v);
50  }
51  }
52  if (origVols.size() != originalShapeCount)
53  {
54  throw std::runtime_error("volume before bool must be provided by the caller or leave it empty");
55  }
56  Standard_Real sumOrigVols = std::accumulate(origVols.cbegin(), origVols.cend(), 0.0);
57 
58  std::vector<Standard_Real> resVols; // result shape volumes
59  Standard_Real allVol = 0;
60  for (const auto& s : resultShapes)
61  {
62  GProp_GProps v_props;
63  BRepGProp::VolumeProperties(s, v_props);
64  auto v = v_props.Mass();
65  resVols.push_back(v);
66  allVol += v;
67  }
68 
69  auto min_result_volume = *std::min_element(resVols.cbegin(), resVols.cend());
70  if (min_result_volume < 0)
71  {
72  VLOG_F(LOGLEVEL_DEBUG, "result solids has minus volume, it is an error");
73  return CollisionType::Error;
74  }
75  auto max_original_volume = *std::max_element(origVols.cbegin(), origVols.cend());
76  // auto min_original_volume = *std::min_element(origVols.cbegin(), origVols.cend());
77 
78  if (resultShapes.size() > originalShapeCount)
79  {
84  auto contact_vol_reltol = 0.01;
85  auto interference_vol_reltol = 0.1; // above that
86  if (floatEqual(allVol, sumOrigVols, contact_vol_reltol))
87  {
88  VLOG_F(LOGLEVEL_DEBUG, "interference volume is small, deemed as face contact by auto fixing");
89  return CollisionType::WeakInterference; // merge the smaller with the interference
90  }
91 
92  else if (floatEqual(allVol, sumOrigVols, interference_vol_reltol))
93  {
94  auto ratio = std::abs((allVol - sumOrigVols) / sumOrigVols);
95  LOG_F(WARNING, "interference ratio is %f, but set as WeakInterference to be fixed", ratio);
96  return CollisionType::WeakInterference; // merge the smaller with the interference
97  }
98  else
99  return CollisionType::Interference;
100  }
101  else
102  {
103  // only surface contact is counted as Contact
104  if (floatEqual(allVol, sumOrigVols, 1e-3))
105  {
106  /* count total face, edge and vertex may also not reliable */
107  Standard_Real totalArea = 0.0;
108  for (const auto& s : origShapes)
109  totalArea += area(s);
110  Standard_Real resultArea = area(mkGFA->Shape());
111  if (floatEqual(totalArea, resultArea)) // not face contact
112  {
113  Standard_Real totalPerimeter = 0.0;
114  for (const auto& s : origShapes)
115  totalPerimeter += perimeter(s);
116  Standard_Real resultPerimeter = perimeter(mkGFA->Shape());
117  if (floatEqual(totalPerimeter, resultPerimeter))
118  return CollisionType::VertexContact;
119  else
120  return CollisionType::EdgeContact;
121  }
122  else
123  {
124  return CollisionType::FaceContact;
125  }
126  }
129  else if (floatEqual(allVol, max_original_volume, 1e-5))
130  {
131  return CollisionType::Enclosure;
132  }
137  else
138  {
139  if (ij.size() == 2)
140  {
141  LOG_F(WARNING, "Collision type unknown for volume change for #%lu , %lu, try BOP check",
142  ij[0], ij[1]);
143  }
144  VLOG_F(LOGLEVEL_DEBUG, "original total vol = %f, fragments total vol = %f \n", sumOrigVols,
145  allVol);
146  VLOG_F(LOGLEVEL_DEBUG, "solids original vols %f, %f\n", origVols[0], origVols[1]);
147  VLOG_F(LOGLEVEL_DEBUG, "fragments count %lu vols %f, %f\n", resVols.size(), resVols[0],
148  resVols[1]);
149 
150  return CollisionType::Unknown;
151  }
152  }
153  }
154  else // no modified solid in general fuse
155  {
156  return CollisionType::NoCollision;
157  }
158  }
159  else // solid count = 0
160  {
161  LOG_F(ERROR, "Error in general fuse, resulted solid number is zero");
162  return CollisionType::Error;
163  }
164  }

167  CollisionInfo CollisionDetector::detectCollision(const TopoDS_Shape& s, const TopoDS_Shape& s2,
168  std::vector<Standard_Real> vols, double theTolerance)
169  {
170  using namespace OccUtils;
171  CollisionType ctype = CollisionType::Error;
172  if (s == s2) // identical obj in memory, inputs are references
173  {
174  ctype = CollisionType::Coincidence;
175  CollisionInfo cinfo = {0, 0, 0.0, ctype}; // index will be set outside by caller
176  return cinfo;
177  }
178 
179  if (vols.size() == 0)
180  {
181  GProp_GProps v_props;
182  BRepGProp::VolumeProperties(s, v_props);
183  vols.push_back(v_props.Mass());
184  BRepGProp::VolumeProperties(s2, v_props);
185  vols.push_back(v_props.Mass());
186  }
187  const std::vector<TopoDS_Shape> shapes = {s, s2};
188  auto mkGFA = std::make_shared<BRepAlgoAPI_BuilderAlgo>();
189  mkGFA->SetNonDestructive(true);
190  mkGFA->SetFuzzyValue(theTolerance);
191  // SetGlue() can accelerate in case of overlapping, but for the case of no interference
192  // mkGFA->SetGlue(BOPAlgo_GlueShift);
193  mkGFA->SetRunParallel(false);
194  mkGFA->SetUseOBB(true);
195  std::exception_ptr eptr;
196  try
197  {
198  generalFuse(shapes, theTolerance, mkGFA);
199  }
200  catch (...)
201  {
202  eptr = std::current_exception();
203  }
204  ctype = calcCollisionType(mkGFA, vols);
205  CollisionInfo cinfo = {0, 0, mkGFA->FuzzyValue(), ctype}; // index will be set outside by caller
206  return cinfo;
207  }

210  bool CollisionDetector::detectCollision(const ItemIndexType i, const ItemIndexType j, bool internalMultiThreading,
211  bool imprinting)
212  {
213  CollisionType ctype = CollisionType::NoCollision;
214 
215  bool completed = false;
216  std::vector<Standard_Real> vols = {(*myGeometryProperties)[i].volume, (*myGeometryProperties)[j].volume};
217  std::vector<TopoDS_Shape> twoShapes = {item(i), item(j)};
218 
219  std::exception_ptr eptr;
220  // bool hasException = false;
221  std::vector<TopoDS_Shape> pieces;
222  // retry by giving bigger tolerance, so tiny interference can be ignored
223  std::shared_ptr<BRepAlgoAPI_BuilderAlgo> pFuser;
224 
225  // it is tricky to whether try fuzzy tol first of zero tol
226  // mastu test data is working either way.
227  // PCarmon's test data has exception if try zero tolerance first
228  std::vector<double> tolerances = {tolerance, 0.0};
229  const size_t NB_RETRY = tolerances.size();
230 
231  // try different tolerance to correct CollisionType::Error
232  for (size_t r = 0; r < NB_RETRY && !completed; r++)
233  {
234  try
235  {
236  auto mkGFA = std::make_shared<BRepAlgoAPI_BuilderAlgo>();
237  mkGFA->SetNonDestructive(true);
238  // mkGFA->SetGlue(BOPAlgo_GlueShift); // quick in case of overlapping only, no interference
239  mkGFA->SetRunParallel(not internalMultiThreading);
240  // mkGFA->SetUseOBB(true); // only if OCCT version is high enough
241  pieces = OccUtils::generalFuse(twoShapes, tolerances[r], mkGFA);
242 
243  pFuser = mkGFA;
244  completed = (pieces.size() == 2);
245  // no need to try again, is that possible to break inside try block?
246  }
247  catch (...)
248  {
249  eptr = std::current_exception();
250  std::vector<ItemIndexType> itemIndices = {i, j};
251  // dealGeneralFuseException(twoShapes, itemIndices);
252  dealGeneralFuseException(pieces, itemIndices);
253  LOG_F(ERROR, "Error during merge and collision type calc for item #%lu `%s`, #%lu `%s`", i,
254  itemName(i).c_str(), j, itemName(j).c_str());
255  ctype = CollisionType::Error; // dump will be done at the end of this function
256  // should not break, loop to see if bigger tolerance solves the GFA exception?
257  }
258 
259  if (pFuser) // this is not valid if exception happened
260  {
261  if (pFuser->HasErrors() and r == NB_RETRY - 1)
262  {
263  // todo: mkGFA->HasError(handle_type) to find out the specific error type
264  LOG_F(ERROR, "general fuse has error for the item pair #%lu `%s`, #%lu `%s`, dump", i,
265  itemName(i).c_str(), j, itemName(j).c_str());
266  auto df_original = generateDumpName("dump_fusionError", {i, j}) + myDumpFileType;
267  OccUtils::saveShape({item(i), item(j)}, df_original);
268  if (pieces.size())
269  {
270  auto df_result = generateDumpName("dump_fusionErrorResult", {i, j}) + myDumpFileType;
271  OccUtils::saveShape(pieces, df_result);
272  }
273  ctype = CollisionType::Error;
274  }
275  else
276  {
277  bool isWeakInterferenceFixed = false;
278  ctype = calcCollisionType(pFuser, vols, {i, j});
279  // if there is volume error, then try if they are not in contact to fix the error
280  // however, it seems always give zero distance, does not help
281  if (ctype >= CollisionType::Error and r == NB_RETRY - 1)
282  {
283  auto ctype_ret = solveErrorByDistanceCheck(i, j);
284  if (ctype_ret <= CollisionType::Clearance)
285  ctype = ctype_ret;
286  }
287  if (ctype == CollisionType::WeakInterference)
288  {
289  auto results = solveWeakInterference(pieces, {}, twoShapes, vols); // modified pieces
290  if (results.size() == 2)
291  {
292  pieces = results; // boundbox, volume and uniqueId has changed after fixing
293  ctype = CollisionType::FaceContact;
294  isWeakInterferenceFixed = true;
295  // todo: save to string and report at the end of processing
296  LOG_F(INFO, "weak interference fixed as face contact for item #%lu `%s`, #%lu `%s`", i,
297  itemName(i).c_str(), j, itemName(j).c_str());
298  }
299  }
300  // must be done after weak interference check and fixing as face contact
301  if (ctype == CollisionType::FaceContact and pieces.size() == 2)
302  {
303  if (not isWeakInterferenceFixed)
304  {
305  // if weak interference is fixed, then both boundbox and volume will change.
306  // so the boundbox change check should be disable or use a very big reltol of 0.3
307  // sequence of input and output shape has been kept, but boundbox may change
308  if (not OccUtils::isBndBoxCoincident(pieces[0], (*myShapeBoundBoxes)[i], 0.3))
309  {
310  VLOG_F(LOGLEVEL_DEBUG, "changed BoundBox for #%lu `%s` when fuse with #%lu", i,
311  itemName(i).c_str(), j);
312  VLOG_F(LOGLEVEL_DEBUG, "face contact detected, write back item #%lu #%lu,", i, j);
313  // volume equalness has been validated in calcCollisionType(), should be equal
314 
315  dumpChangedBoundbox(i, pieces[0]);
316  }
317  if (not OccUtils::isBndBoxCoincident(pieces[1], (*myShapeBoundBoxes)[j], 0.3))
318  {
319  VLOG_F(LOGLEVEL_DEBUG, "changed BoundBox for #%lu `%s` when fuse with #%lu", j,
320  itemName(j).c_str(), i);
321  // dumpChangedBoundbox(j, pieces[1]); // it must has been dump by other operation
322  }
323  }
324 
325  if (imprinting) // code for Imprinter only
326  {
327  setItem(i, pieces[0]); // should also update boundbox and recalc volume
328  setItem(j, pieces[1]);
329  }
330  myAdjacencyMatrix.insertAt(i, j, true);
331  myAdjacencyMatrix.insertAt(j, i, true);
332  }
333  else if (ctype == CollisionType::NoCollision)
334  {
335  calcClearance(i, j); // also write myCollisionInfos
336  }
337  else
338  {
339  // do nothing
340  }
341  }
342  }
343  if (ctype < CollisionType::Error) // does not need to retry
344  break;
345  } // end of retry
346 
347  // here CollisionType::FaceContact must be saved to help resolve collision!
348  if (ctype >= CollisionType::FaceContact) // vertex or edge contact is not reported
349  {
350  CollisionInfo info = {i, j, tolerance, ctype};
351  myCollisionInfos[i].push_back(std::make_pair(j, info));
352  CollisionInfo info_j = {j, i, tolerance, ctype};
353  myCollisionInfos[j].push_back(std::make_pair(i, info_j));
354  // report collision will be done in postProcess()
355  }
356  else if (ctype >= CollisionType::Error)
357  {
358  auto df_original = generateDumpName("dump_ctypeErrorOriginal", {i, j}) + myDumpFileType;
359  OccUtils::saveShape({item(i), item(j)}, df_original);
360  }
361  // VLOG_F(LOGLEVEL_DEBUG, "contact type is %i solid pair #%lu `%s`, #%lu `%s`", ctype, i,
362  // itemName(i).c_str(), j, itemName(j).c_str()); // disable this debug, using myCollisionInfos.json
363  return completed && ctype == CollisionType::FaceContact;
364  }

389  }

397  }

408  myCollisionInfos[j].push_back(std::make_pair(i, info_j));

415  VLOG_F(LOGLEVEL_DEBUG,

465  std::vector<TopoDS_Shape> CollisionDetector::solveWeakInterference(const std::vector<TopoDS_Shape> resultsPieces,
466  std::vector<Standard_Real> resultsVols,
467  const std::vector<TopoDS_Shape>, // origItems
468  const std::vector<Standard_Real> origVols)
469  {
470  // std::shared_ptr<const BRepAlgoAPI_BuilderAlgo> mkGFA
471  if (resultsVols.size() == 0) // empty, default parameter value
472  {
473  for (const auto& s : resultsPieces)
474  {
475  GProp_GProps v_props;
476  BRepGProp::VolumeProperties(s, v_props);
477  auto v = v_props.Mass();
478  resultsVols.push_back(v);
479  }
480  }
481 
483  auto iBig = (origVols[0] > origVols[1]) ? 0 : 1;
484  auto iSmall = (origVols[0] < origVols[1]) ? 0 : 1;
485  // choose a strategy: merge to the biggest or the smallest result fragament?
486  auto iKept = iSmall;
488  auto iKept_result = shapeMatch(origVols[iKept], resultsPieces, resultsVols);
489 
490  std::vector<TopoDS_Shape> results;
491  TopoDS_Shape theKept = resultsPieces[iKept_result];
492  TopoDS_Shape theOtherShape;
493  auto restPieces = std::vector<TopoDS_Shape>();
494  for (size_t i = 0; i < resultsPieces.size(); i++)
495  {
496  if (i != iKept_result)
497  restPieces.push_back(resultsPieces[i]);
498  }
499  if (resultsPieces.size() > 2)
500  {
501  try
502  {
503  OCC_CATCH_SIGNALS
504  theOtherShape = OccUtils::fuseShape(restPieces);
505  }
506  catch (const Standard_Failure& fail)
507  {
508  VLOG_F(LOGLEVEL_DEBUG, "OCCT Standard_Failure %s in shape fusion in weak inference fixing",
509  fail.GetMessageString());
510  theOtherShape = restPieces[0];
511  }
512  catch (const std::exception& e)
513  {
514  VLOG_F(LOGLEVEL_DEBUG, "Exception %s in shape fusion in weak inference fixing", e.what());
515  theOtherShape = restPieces[0];
516  }
517  }
518  else
519  {
520  size_t iOther_result = iKept_result == 1 ? 0 : 1;
521  theOtherShape = resultsPieces[iOther_result];
522  }
523 
524  // merge all the other smaller, or using the original smaller volume
525  if (iKept == 0)
526  {
527  results.push_back(theKept);
528  results.push_back(theOtherShape);
529  }
530  else
531  {
532  results.push_back(theOtherShape);
533  results.push_back(theKept);
534  }
535  return results;
536  }

655  void CollisionDetector::processCollisionInfo(const ItemIndexType i)
656  {
657  if (itemSuppressed(i))
658  return; // has been resolved by suppressing this item before call this function
659 
660  const auto& row = myCollisionInfos[i];
661  if (row.size() == 0) // NoCollision is not saved
662  {
663  // one body without contact/interference with other bodies, suppress it
664  if (suppressFloating)
665  {
666  suppressItem(i, ShapeErrorType::FloatingShape);
667  // VLOG_F(LOGLEVEL_DEBUG, "Item #%lu `%s` suppressed as it is floating, no contact with other", i,
668  // itemName(i).c_str());
669  }
670  else
671  {
672  // suppress this debug information, too noisy, analyzeProcessedResult.py can retrieve this info
673  // VLOG_F(LOGLEVEL_DEBUG, "Item #%lu `%s` in NOT suppressed, although a floating shape ", i,
674  // itemName(i).c_str());
675  }
676  // todo: report to the operator but does not need response;
677  }
678  else if (row.size() == 1)
679  {
680  const auto& info = row[0].second;
681  const CollisionType ctype = info.type;
682  const auto j = info.second;
683  // remove the smaller, or keep the one with more contactCount
684  if (ctype == CollisionType::Enclosure)
685  {
686  auto vol_i = (*myGeometryProperties)[i].volume;
687  auto vol_j = (*myGeometryProperties)[j].volume;
688  auto k = (vol_i < vol_j) ? i : j;
689  if (not itemSuppressed(k))
690  {
691  suppressItemImpl(k, CollisionType::Enclosure);
692  }
693  }
694  else if (ctype == CollisionType::Error)
695  {
696  if (not itemSuppressed(i))
697  {
698  suppressItemImpl(i, ctype);
699  }
700  // notifyCollision(info, "Error: during general fuse to detect collision type");
701  }
702  else if (ctype == CollisionType::Clearance)
703  {
704  // todo: assembly is not yet done, it should be in contact with at least one
705  // notifyCollision(info, "Error: this part has a small clearance with only one neighbour");
706  }
707  else if (ctype == CollisionType::Interference)
708  {
709  // the other has no other interference, and total items > 2
710  // if (myCollisionInfos[j].size() > 1 and countType(j, CollisionType::FaceContact) > 0)
711  if (not itemSuppressed(i))
712  {
713  suppressItemImpl(i, ctype);
714  }
715  }
716  else if (ctype == CollisionType::Coincidence)
717  {
718  // item i and j are both floating, on relation with other items
719  if (not itemSuppressed(i))
720  {
721  suppressItemImpl(i, ctype);
722  }
723  // LOG_F(info, "Error: both shapes with coincidence are not in contact with other");
724  }
725  else
726  {
727  return; // do nothing
728  }
729  }
730  else // > 1, interference, coincidence, enclosure has been dealt
731  {
732  // three or more body interference not checked.
733  if (hasAtLeastType(i, CollisionType::Interference))
734  {
735  // Enclosure has been solved, here solves CollisionType::Coincidence
736  bool coincidenceResolved = false;
737  for (const auto& p : row)
738  {
739  const auto& cinfo = p.second;
740  auto j = cinfo.second;
741  if (cinfo.type == CollisionType::Coincidence)
742  {
743  if (i < j) // remove either is fine, but avoid removing twice!
744  suppressItem(i, ShapeErrorType::Coincidence);
745  coincidenceResolved = true;
746  }
747  if (cinfo.type == CollisionType::Enclosure)
748  {
749  // the smaller volume must only 1 relationship with other,
750  // so it should have been suppressed
751  coincidenceResolved = true;
752  }
753  }
754 
755  // if one body interferes/has error with the other,
756  // the less total interference should be kept, disable the item with bigger Error count
757  // in case of CollisionType::Unknown, just leave it
758  resolveCollisionError(i, CollisionType::Interference);
759  resolveCollisionError(i, CollisionType::Error);
760  }
761  }

794  bool CollisionDetector::checkCollisionResolution()
795  {
796  ItemIndexType floating = 0;
797  // unsigned int floating = 0;
798  std::map<ItemIndexType, CollisionType> unresolvedItems;
799  const bool ignore_unknown = parameter<bool>("ignoreUnknownCollisionType");
800  for (std::size_t i = 0; i < myInputData->itemCount(); i++)
801  {
802  const auto& row = myCollisionInfos[i];
803  if (row.size() == 0)
804  floating++;
805  for (const auto& p : row)
806  {
807  const auto& cinfo = p.second;
808  auto j = cinfo.second;
809  // ignore CollisionType::Unknown, it should have been dumped during collisionType calc
810  if (cinfo.type == CollisionType::Unknown)
811  {
812  if (not(itemSuppressed(i) or itemSuppressed(j)) and (not ignore_unknown))
813  {
814  unresolvedItems.insert(std::make_pair(i, cinfo.type));
815  unresolvedItems.insert(std::make_pair(j, cinfo.type));
816  }
817  }
818  if (cinfo.type >= CollisionType::Interference and cinfo.type < CollisionType::Unknown)
819  {
820  if (not(itemSuppressed(i) or itemSuppressed(j)) and suppressErroneous)
821  {
822  unresolvedItems.insert(std::make_pair(i, cinfo.type));
823  unresolvedItems.insert(std::make_pair(j, cinfo.type));
824  }
825  }
826  }
827  }
828  for (const auto& k : unresolvedItems)
829  {
830  notifyUnresolvedCollision(k.first, k.second);
831  }
832  if (floating > 0)
833  LOG_F(WARNING, "floating item count is %lu out of total %lu", floating, itemCount());
834  return unresolvedItems.size() == 0;
835  }
