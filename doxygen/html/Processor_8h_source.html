182  virtual void report()
183  {
184  if (!hasParameter("report"))
185  return;
186  Information report;
187  const std::size_t NItems = myItemReports.size();
188  size_t reportItemCount = 0;
189  for (std::size_t i = 0; i < NItems; i++)
190  {
191  if (hasItemReport(i))
192  {
193  const std::string& s = itemReport(i).str();
194  // report[itemName(i)] = std::string(s);
195  report[std::to_string(i)] = std::string(s);
196  reportItemCount++;
197  }
198  }
199  if (reportItemCount)
200  {
201  auto outFile = dataStoragePath(parameter<std::string>("report"));
202  std::ofstream o(outFile);
203  o << report;
204  }
205  // todo: instanceID unique name to distinguish instances, or just
206  // (*myInfo)[className()] = report;
207  }
