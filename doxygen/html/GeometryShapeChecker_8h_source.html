63  virtual void processItem(const ItemIndexType i) override final
64  {
65  if (itemSuppressed(i))
66  return;
67  const TopoDS_Shape& aShape = item(i);
68  std::string err = checkShape(aShape);
69  auto ssp = std::make_shared<std::stringstream>();
70  if (err.size() > 0)
71  *ssp << err;
72 
73  // if basic checkShape has error, then BOPCheck must have fault?
74  if ( // err.size() == 0 && skip BOP check if preliminary check has failed?
75  checkingBooleanOperation) // Two common not serious errors are skipped in this BOP check
76  {
77  bool hasBOPFault = false;
78  try
79  {
80  hasBOPFault = BOPSingleCheck(i, *ssp); // why most of shape return true even no error msg???
81  }
82  catch (const std::exception& e)
83  {
84  hasBOPFault = true;
85  LOG_F(ERROR, "BOP check has std::exception %s for item %lu ", e.what(), i);
86  }
87  catch (...)
88  {
89  hasBOPFault = true;
90  LOG_F(ERROR, "BOP check has other exception for item %lu ", i);
91  }
92 
93  if (hasBOPFault && ssp->str().size() > 0)
94  {
95  auto df = generateDumpName("dump_BOPCheckFailed", {i}) + ".brep";
96  OccUtils::saveShape({item(i)}, df);
97  if (suppressBOPCheckFailed)
98  {
99  LOG_F(ERROR, "BOP check found fault for item %lu, so suppress this item", i);
100  suppressItem(i);
101  }
102  else
103  {
104  LOG_F(ERROR, "BOP check found fault for item %lu, ignore this error", i);
105  }
106  }
107  if (ssp->str().size() > 0)
108  setItemReport(i, ssp);
109  }
110  }

294  bool BOPSingleCheck(const ItemIndexType i, std::stringstream& ss)
295  {
296  const TopoDS_Shape& aShape = item(i);
297  bool runSingleThreaded = true; // parallel is done on solid shape level for PPP, not on subshape level
298 
299  bool argumentTypeMode = true;
300  bool selfInterMode = true; // self interference, for single solid should be no such error
301  bool smallEdgeMode = true; // only needed for de-feature?
302 
303  bool continuityMode = false; // BOPAlgo_GeomAbs_C0, it should not cause BOP failure?
304  bool tangentMode = false; // not implemented in OpenCASCADE
305 
306  bool rebuildFaceMode = true;
307  bool mergeVertexMode = true; // leader to `BOPAlgo_IncompatibilityOfEdge`
308  bool mergeEdgeMode = true;
309  bool curveOnSurfaceMode = false; // BOPAlgo_InvalidCurveOnSurface: tolerance compatibility check
310 
311  // FreeCAD develper's note: I don't why we need to make a copy, but it doesn't work without it.
313  TopoDS_Shape BOPCopy = BRepBuilderAPI_Copy(aShape).Shape();
314  BOPAlgo_ArgumentAnalyzer BOPCheck;
315 
316  // BOPCheck.StopOnFirstFaulty() = true; //this doesn't run any faster but gives us less results.
317  // BOPCheck.SetFuzzyValue();
318  BOPCheck.SetShape1(BOPCopy);
319  // BOPCheck.SetOperation(); // by default, set to BOPAlgo_UNKNOWN
320  // all settings are false by default. so only turn on what we want.
321  BOPCheck.ArgumentTypeMode() = argumentTypeMode;
322  BOPCheck.SelfInterMode() = selfInterMode;
323  BOPCheck.SmallEdgeMode() = smallEdgeMode;
324  BOPCheck.RebuildFaceMode() = rebuildFaceMode;
325 #if OCC_VERSION_HEX >= 0x060700
326  BOPCheck.ContinuityMode() = continuityMode;
327 #endif
328 #if OCC_VERSION_HEX >= 0x060900
329  BOPCheck.SetParallelMode(!runSingleThreaded);
330  BOPCheck.SetRunParallel(!runSingleThreaded);
331 
332  BOPCheck.TangentMode() = tangentMode; // these 4 new tests add about 5% processing time.
333  BOPCheck.MergeVertexMode() = mergeVertexMode; // will it modify the shape to check?
334  BOPCheck.MergeEdgeMode() = mergeEdgeMode;
335  BOPCheck.CurveOnSurfaceMode() = curveOnSurfaceMode;
336 #endif
337 
338  BOPCheck.Perform(); // this perform() has internal try-catch block
339  if (BOPCheck.HasFaulty() and savingSubshape)
340  {
341  const BOPAlgo_ListOfCheckResult& BOPResults = BOPCheck.GetCheckResult();
342  BOPAlgo_ListIteratorOfListOfCheckResult BOPResultsIt(BOPResults);
343  for (size_t j = 0; BOPResultsIt.More(); BOPResultsIt.Next(), j++)
344  {
345  const BOPAlgo_CheckResult& current = BOPResultsIt.Value();
346  if (current.GetCheckStatus() != BOPAlgo_CheckUnknown)
347  {
348  ss << ";" << getBOPCheckStatusName(current.GetCheckStatus());
349  dumpSubshape(i, j, current);
350  }
351  }
352  return true; // has failure
353  }
354  return false; // no fault
355  }

359  const auto& faultyShapes1 = result.GetFaultyShapes1();

365  const auto& subshape = faultyShapes1It.Value();
366  OccUtils::saveShape(subshape, df);
367  }
